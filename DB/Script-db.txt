CREATE TABLE ROL (
    id_rol SERIAL PRIMARY KEY,
    nombre_rol VARCHAR(50) NOT NULL,
    descripcion VARCHAR(200)
);

-- 2. Tabla PERMISO
CREATE TABLE PERMISO (
    id_permiso SERIAL PRIMARY KEY,
    nombre_permiso VARCHAR(100) NOT NULL
);

-- 3. Tabla ROL_PERMISO (Intermedia)
CREATE TABLE ROL_PERMISO (
    id_rol INT NOT NULL REFERENCES ROL(id_rol) ON DELETE CASCADE,
    id_permiso INT NOT NULL REFERENCES PERMISO(id_permiso) ON DELETE CASCADE,
    PRIMARY KEY (id_rol, id_permiso)
);

-- 4. Tabla USUARIO
CREATE TABLE USUARIO (
    id_usuario SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    correo VARCHAR(150) NOT NULL UNIQUE,
    contraseña_hash VARCHAR(255) NOT NULL,
    fecha_registro TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    activo BOOLEAN NOT NULL DEFAULT TRUE, 
    id_rol INT NOT NULL REFERENCES ROL(id_rol)
);

-- 5. Tabla TOKENS_SESION
CREATE TABLE TOKENS_SESION (
    id_sesion SERIAL PRIMARY KEY,
    token TEXT NOT NULL,
    fecha_inicio TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    fecha_fin TIMESTAMP,
    id_usuario INT NOT NULL REFERENCES USUARIO(id_usuario) ON DELETE CASCADE
);

-- 6. Tabla HUB
CREATE TABLE HUB (
    id_hub SERIAL PRIMARY KEY,
    ip_local VARCHAR(15) NOT NULL,
    version_firmware VARCHAR(20) NOT NULL,
    estado_conexion VARCHAR(50) NOT NULL
);

-- 7. Tabla USUARIO_HUB (Intermedia)
CREATE TABLE USUARIO_HUB (
    id_usuario INT NOT NULL REFERENCES USUARIO(id_usuario) ON DELETE CASCADE,
    id_hub INT NOT NULL REFERENCES HUB(id_hub) ON DELETE CASCADE,
    PRIMARY KEY (id_usuario, id_hub)
);

-- 8. Tabla DISPOSITIVO
CREATE TABLE DISPOSITIVO (
    id_dispositivo SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    tipo VARCHAR(50) NOT NULL,
    ubicacion VARCHAR(100) NOT NULL,
    estado VARCHAR(50) NOT NULL,
    id_hub INT NOT NULL REFERENCES HUB(id_hub)
);

-- 9. Tabla ESCENA
CREATE TABLE ESCENA (
    id_escena SERIAL PRIMARY KEY,
    nombre VARCHAR(80) NOT NULL,
    descripcion VARCHAR(200),
    activa BOOLEAN NOT NULL DEFAULT TRUE,
    id_hub INT NOT NULL REFERENCES HUB(id_hub)
);

-- 10. Tabla ACCION_ESCENA
CREATE TABLE ACCION_ESCENA (
    id_accion SERIAL PRIMARY KEY,
    id_escena INT NOT NULL REFERENCES ESCENA(id_escena) ON DELETE CASCADE,
    id_dispositivo INT NOT NULL REFERENCES DISPOSITIVO(id_dispositivo),
    comando VARCHAR(100) NOT NULL,
    parametro VARCHAR(100)
);

-- 11. Tabla TAREAS_PROGRAMADAS
CREATE TABLE TAREAS_PROGRAMADAS (
    id_tarea SERIAL PRIMARY KEY,
    expresion_cron VARCHAR(50) NOT NULL,
    descripcion VARCHAR(200),
    activa BOOLEAN NOT NULL DEFAULT TRUE,
    id_escena INT REFERENCES ESCENA(id_escena) ON DELETE SET NULL,
    id_dispositivo INT REFERENCES DISPOSITIVO(id_dispositivo) ON DELETE SET NULL
);

-- 12. Tabla ALERTA
CREATE TABLE ALERTA (
    id_alerta SERIAL PRIMARY KEY,
    tipo VARCHAR(50) NOT NULL,
    mensaje VARCHAR(255) NOT NULL,
    fecha_hora TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    leida BOOLEAN NOT NULL DEFAULT FALSE, 
    id_dispositivo INT NOT NULL REFERENCES DISPOSITIVO(id_dispositivo),
    id_usuario INT NOT NULL REFERENCES USUARIO(id_usuario)
);

-- 13. Tabla PREFERENCIAS_NOTIFICACION
CREATE TABLE PREFERENCIAS_NOTIFICACION (
    id_preferencia SERIAL PRIMARY KEY,
    id_usuario INT NOT NULL REFERENCES USUARIO(id_usuario) ON DELETE CASCADE,
    tipo_alerta VARCHAR(50) NOT NULL,
    canal_push BOOLEAN NOT NULL DEFAULT TRUE,
    canal_email BOOLEAN NOT NULL DEFAULT TRUE
);

-- 14. Tabla HISTORIAL_AUDITORIA
CREATE TABLE HISTORIAL_AUDITORIA (
    id_auditoria SERIAL PRIMARY KEY,
    accion VARCHAR(100) NOT NULL,
    tabla_afectada VARCHAR(50),
    id_registro_afectado INT,
    fecha_hora TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    id_usuario INT REFERENCES USUARIO(id_usuario) ON DELETE SET NULL,
    detalles_json TEXT 
);
